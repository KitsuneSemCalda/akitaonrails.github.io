{{- /* Helper to be a strong typed integer param with hugo limitations */}} {{- $val := index . 0 }} {{- $default := index . 1 }} {{- if and $val (ne $val "") }} {{- /* Force cast para inteiro puro, se falhar, fallback no default */ -}} {{- $intVal := 0 -}} {{- with (int $val) }} {{- $intVal = . -}} {{- else }} {{- $intVal = $default -}} {{- end }} {{- $intVal -}} {{- else }} {{- $default -}} {{- end }}
